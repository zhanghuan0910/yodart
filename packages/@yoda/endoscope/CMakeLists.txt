cmake_minimum_required(VERSION 3.0)
project(yoda-endoscope CXX)
set(CMAKE_CXX_STANDARD 11)

file(GLOB YODA_ENDOSCOPE_SRC *.js *.json)
file(GLOB YODA_ENDOSCOPE_EXPORTER_SRC exporter/*.js)
file(GLOB YODA_ENDOSCOPE_METRIC_SRC metric/*.js)

install(FILES ${YODA_ENDOSCOPE_SRC} DESTINATION ${CMAKE_INSTALL_DIR})
install(FILES ${YODA_ENDOSCOPE_EXPORTER_SRC} DESTINATION ${CMAKE_INSTALL_DIR}/exporter)
install(FILES ${YODA_ENDOSCOPE_METRIC_SRC} DESTINATION ${CMAKE_INSTALL_DIR}/metric)
